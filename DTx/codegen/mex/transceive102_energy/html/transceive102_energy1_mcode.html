1function [dr,ns] = transceive102_energy(d2s,ft,txGain,rxGain,centerFreqTx,centerFreqRx,intFactor,decFactor,swapFreqFlag)
 2persistent hrx htx; 
 3dr = complex(zeros(1408,1));
 4ns = uint32(0);
 5if isempty(htx)
 6    htx = comm.SDRuTransmitter('CenterFrequency',centerFreqTx,'Gain',txGain, ...
 7        'InterpolationFactor',intFactor,'IPAddress','192.168.10.2', ...
 8        'LocalOscillatorOffset',0);
 9end
10if isempty(hrx)
11    hrx = comm.SDRuReceiver('CenterFrequency',centerFreqRx, ...
12        'DecimationFactor',decFactor,'FrameLength',1408,'Gain',rxGain, ...
13        'IPAddress','192.168.10.2','OutputDataType','double', ...
14        'SampleRate',1e5);
15end
16
17%listening mode:
18if abs(centerFreqTx-centerFreqRx)>0
19    %if Rx and Tx is different, switch for Listening mode
20    if swapFreqFlag
21        hrx.CenterFrequency = centerFreqTx;
22    else
23        hrx.CenterFrequency = centerFreqRx;
24    end
25end
26
27if (ft)
28    release(hrx);
29    release(htx);
30else
31    step(htx,d2s);
32    while (ns < uint32(1))
33        [dr,ns] = step(hrx); 
34    end    
35end
36return;
37end
