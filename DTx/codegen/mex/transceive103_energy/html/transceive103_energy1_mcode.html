1function [dr,ns] = transceive103_energy(d2s,ft,txGain,rxGain,centerFreqTx,centerFreqRx,intFactor,decFactor,flag)
 2persistent hrx htx; 
 3dr = complex(zeros(1408,1));
 4ns = uint32(0);
 5if isempty(htx)
 6    htx = comm.SDRuTransmitter('CenterFrequency',centerFreqTx,'Gain',txGain, ...
 7        'InterpolationFactor',intFactor,'IPAddress','192.168.10.3', ...
 8        'LocalOscillatorOffset',0);
 9end
10if isempty(hrx)
11    hrx = comm.SDRuReceiver('CenterFrequency',centerFreqRx, ...
12        'DecimationFactor',decFactor,'FrameLength',1408,'Gain',rxGain, ...
13        'IPAddress','192.168.10.3','OutputDataType','double', ...
14        'SampleRate',2e5);
15end
16
17if flag
18    hrx.CenterFrequency = centerFreqTx;
19end
20
21if (ft)
22    release(hrx);
23    release(htx);
24else

26    while (ns < uint32(1))
27        [dr,ns] = step(hrx); 
28    end    
29end

31end
