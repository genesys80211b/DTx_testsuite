1function reportSDRuStatus(errStatus, errMsg, radioID, reportMethod)
  2%reportSDRuStatus SDRu status reporter
  3%   reportSDRuStatus(STATUS,MSG,IP,METHOD) reports the SDRu status based on
  4%   the STATUS input. STATUS is a UsrpErrorCapiEnumT type. IP is the IP
  5%   address of the USRP(R) radio. METHOD is a UsrpReportMethodEnumT type and
  6%   can be Warning, Error , or None.
  7%
  8%   USRP(R) is a trademark of National Instruments Corp.
  9
 10%   Copyright 2012-2014 The MathWorks, Inc.
 11
 12%#codegen
 13
 14  coder.internal.errorIf(...
 15    ~isa(errStatus, 'UsrpErrorCapiEnumT') || ~isscalar(errStatus), ...
 16    'sdru:reportSDRuStatus:InvalidErrorStatus')
 17
 18  coder.internal.errorIf(~ischar(errMsg), ...
 19    'sdru:reportSDRuStatus:InvalidErrorMessage', errMsg);
 20
 21  coder.internal.errorIf(~ischar(radioID), ...
 22    'sdru:reportSDRuStatus:InvalidIPAddress');
 23  coder.internal.errorIf(~isrow(radioID), ...
 24    'sdru:reportSDRuStatus:InvalidIPAddress');
 25
 26  coder.internal.errorIf(...
 27    ~isa(reportMethod, 'UsrpReportMethodEnumT') || ~isscalar(reportMethod), ...
 28    'sdru:reportSDRuStatus:InvalidReportMethod')
 29
 30  if isempty(strfind(radioID, '.'))
 31      errstr = 'with serial number';
 32  else
 33      errstr = 'at address';
 34  end
 35      
 36  switch errStatus
 37    case UsrpErrorCapiEnumT.UsrpDriverNotCompatible
 38      msgID = 'sdru:reportSDRuStatus:NotCompatible';
 39      switch reportMethod
 40        case UsrpReportMethodEnumT.Warning
 41          coder.internal.warning(msgID, radioID, getSDRuDriverVersion);
 42        case UsrpReportMethodEnumT.Error
 43          if isempty(coder.target)
 44            coder.internal.errorIf(...
 45              errStatus == UsrpErrorCapiEnumT.UsrpDriverNotCompatible, ...
 46              msgID, radioID, getSDRuDriverVersion);
 47          else
 48            coder.internal.errorIf(...
 49              errStatus == UsrpErrorCapiEnumT.UsrpDriverNotCompatible, ...
 50              msgID, radioID, 'returned by ''getSDRuDriverVersion'' function');
 51          end
 52      end
 53    case UsrpErrorCapiEnumT.UsrpDriverNotResponding
 54      msgID = 'sdru:reportSDRuStatus:NotResponding';
 55      switch reportMethod
 56        case UsrpReportMethodEnumT.Warning
 57          coder.internal.warning(msgID, radioID);
 58        case UsrpReportMethodEnumT.Error
 59          if strcmp(errstr, 'with serial number')
 60            coder.internal.errorIf(...
 61              errStatus == UsrpErrorCapiEnumT.UsrpDriverNotResponding, ...
 62              'sdru:reportSDRuStatus:UnknownStatus', ...
 63              ['Device with serial number ''' radioID ''' is not responding. Attach a device with this serial number, or continue working in offline mode']);
 64          else
 65            coder.internal.errorIf(...
 66              errStatus == UsrpErrorCapiEnumT.UsrpDriverNotResponding, ...
 67              msgID, radioID);
 68          end
 69      end
 70    case UsrpErrorCapiEnumT.UsrpDriverRxBusy
 71      msgID = 'sdru:reportSDRuStatus:RxBusy';
 72      switch reportMethod
 73        case UsrpReportMethodEnumT.Warning
 74          coder.internal.warning(msgID, radioID);
 75        case UsrpReportMethodEnumT.Error
 76          coder.internal.errorIf(...
 77            errStatus == UsrpErrorCapiEnumT.UsrpDriverRxBusy, ...
 78            msgID, radioID);
 79      end
 80    case UsrpErrorCapiEnumT.UsrpDriverTxBusy
 81      msgID = 'sdru:reportSDRuStatus:TxBusy';
 82      switch reportMethod
 83        case UsrpReportMethodEnumT.Warning
 84          coder.internal.warning(msgID, radioID);
 85        case UsrpReportMethodEnumT.Error
 86          coder.internal.errorIf(...
 87            errStatus == UsrpErrorCapiEnumT.UsrpDriverTxBusy, ...
 88            msgID, radioID);
 89      end
 90    case UsrpErrorCapiEnumT.UsrpDriverBusy
 91      msgID = 'sdru:reportSDRuStatus:Busy';
 92      switch reportMethod
 93        case UsrpReportMethodEnumT.Warning
 94          coder.internal.warning(msgID, radioID);
 95        case UsrpReportMethodEnumT.Error
 96          coder.internal.errorIf(...
 97            errStatus == UsrpErrorCapiEnumT.UsrpDriverBusy, ...
 98            msgID, radioID);
 99      end
100    case UsrpErrorCapiEnumT.UsrpDriverError
101      msgID = 'sdru:reportSDRuStatus:UnknownStatus';
102      switch reportMethod
103        case UsrpReportMethodEnumT.Warning
104          coder.internal.warning(msgID, errMsg)
105        case UsrpReportMethodEnumT.Error
106          coder.internal.errorIf(...
107            errStatus == UsrpErrorCapiEnumT.UsrpDriverError, msgID, errMsg)
108      end
109  end
110end
