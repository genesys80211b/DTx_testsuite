1function flag = checkIPAddressFormat(aAddrs, addrsName)
 2%checkIPAddressFormat IP address format checker
 3%
 4%   Warning: This function is an internal MathWorks function and may be removed
 5%   in a future release.
 6%
 7%   checkIPAddressFormat(ADDRS, NAME) returns true if the ADDRS is a valid
 8%   dotted quad IP address. Otherwise, it errors out and prompts that variable
 9%   NAME is not a valid IP address.
10%
11%   % Example:
12%   %   Check if the string '192.168.10.a' is a valid IP address for variable
13%   %   IPAddress
14%   checkIPAddressFormat('192.168.10.a', 'IPAddress')
15
16%   Copyright 2012 The MathWorks, Inc.
17%#codegen
18
19nargchk(2,2,nargin,'struct');  %#ok
20
21coder.internal.errorIf(~ischar(aAddrs), ...
22    'sdru:checkIPAddressFormat:InvalidIPAddress', addrsName);
23coder.internal.errorIf(~isrow(aAddrs), ...
24    'sdru:checkIPAddressFormat:InvalidIPAddress', addrsName);
25
26coder.internal.errorIf(~ischar(addrsName), ...
27    'sdru:checkIPAddressFormat:InvalidIPAddressName');
28coder.internal.errorIf(~isrow(addrsName), ...
29    'sdru:checkIPAddressFormat:InvalidIPAddressName');
30
31% Look for three dots
32dotIndices = find(aAddrs == '.'); % Used instead of findstr since
33% findstr does not generate code
34coder.internal.errorIf(length(dotIndices) ~= 3, ...
35  'sdru:checkIPAddressFormat:InvalidIPAddress', addrsName);
36
37% Check if all the elements other than the dots are numbers
38num1 = aAddrs(1:dotIndices(1)-1);
39num2 = aAddrs(dotIndices(1)+1:dotIndices(2)-1);
40num3 = aAddrs(dotIndices(2)+1:dotIndices(3)-1);
41num4 = aAddrs(dotIndices(3)+1:length(aAddrs));
42
43checkNumeric(int16(num1), addrsName);
44checkNumeric(int16(num2), addrsName);
45checkNumeric(int16(num3), addrsName);
46checkNumeric(int16(num4), addrsName);
47
48flag = true;
49end
