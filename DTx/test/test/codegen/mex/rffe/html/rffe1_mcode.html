1function dfr = rffe(df,ft,cas,cau,cef)
 2% RFFE: Radio Freq Front End: AGC's, Freq Compensates, & RCRF's Input Data
 3% Function Arguments: 
 4% df:  This Data Frame: Raw data taken from USRP Rx Buffer (length 1408)
 5% dfr: This Data Frame Recovered: Data after AGC, FOC & RCRF (length 704)
 6% ft:  Flag Terminal: If true, specifies to release all System objects
 7% prm: DBPSK Transceiver Parameter Structure
 8
 9% ha: Handle to Automatic Gain Control (AGC) System object
10persistent ha;
11% he: Handle to Coarse Frequency Estimator (CFE) System object
12persistent he;
13% hf: Handle to Frequency Offset Compensator (FOC) System object
14persistent hf;
15% hrr: Handle to Raised Cosine Receive Filter (RCRF) System object
16persistent hrr;
17% Initialize Persistent Data: Only on first call to RFFE()
18if isempty(ha)
19    ha = comm.AGC('DetectorMethod','Rectifier','LoopMethod','Logarithmic', ...
20        'MaximumGain',30,'StepSize',cas,'UpdatePeriod',cau);
21end % END IF ISEMPTY(HA)
22if isempty(he)
23    he = comm.PSKCoarseFrequencyEstimator('Algorithm','FFT-based', ...
24        'FrequencyResolution',cef,'ModulationOrder',2,'SampleRate',1e5);         
25end % END IF ISEMPTY(HC)
26if isempty(hf)
27    hf = comm.PhaseFrequencyOffset('PhaseOffset',0, ...
28        'FrequencyOffsetSource','Input port','SampleRate',1e5);
29end % END IF ISEMPTY(HF)
30if isempty(hrr)
31    hrr = comm.RaisedCosineReceiveFilter('DecimationFactor',2,...
32        'DecimationOffset',0,'FilterSpanInSymbols',8,'Gain',1,...
33        'InputSamplesPerSymbol',2,'RolloffFactor',0.3,'Shape','Square root');
34end % IF ISEMPTY(HRR)
35% Local data preallocation
36dfr = complex(zeros(704,1));
37ofs = double(0.0); %#ok<NASGU>
38tmp = complex(zeros(1408,1)); %#ok<NASGU>
39if ft
40    % Release System objects: only on final call to RFFE()
41    release(ha);
42    release(he);
43    release(hf);
44    release(hrr);
45    %clear('ha','he','hf','hrr'); %Not supported for code generation
46else
47    % Automatic Gain Control (AGC)
48    tmp = step(ha,df);
49    % Coarse Frequency Offset Estimation (CFE)
50    ofs = step(he,tmp);
51    % Frequency Offset Compensation (FOC)
52    tmp = step(hf,tmp,-ofs);
53    % Raised Cosine Receive Filtration (RCRF)
54    dfr = step(hrr,tmp);
55end % IF FT
56end % FUNCTION RFFE
